library(dplyr)
library(ggplot2)

# Get mean predicted probability by age separately for each vaccine group
temp1 <- dat_2 %>%

  group_by(age) %>%
  summarise(prob_unvax = mean(prob_covidvisit))

  filter(vaccine == 1) %>%
  group_by(age) %>%
  summarise(prob_vax = mean(prob_covidvisit))

# Merge and calculate RR and RD
rrplot <- inner_join(temp1, temp2, by = "age") %>%
  mutate(
    RR = prob_vax / prob_unvax,
    RD = prob_vax - prob_unvax

ggplot(rrplot, aes(x = age)) +
  geom_point(aes(y = RR, color = "RR")) +
  geom_point(aes(y = RD, color = "RD")) +
  labs(
    x     = "Age (years)",

    color = "Measure",
    title = "RR and RD by age: vaccinated vs unvaccinated"
